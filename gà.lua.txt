        while wait(0.5) do
            if not Config.AutoFarmLevel then break end

                    VirtualInput:SendKeyEvent(false, "V", false, game)
                end
            end
        end
    end)
end

            if not Config.AutoFarmBone then break end

                if string.find(enemy.Name:lower(), "skeleton") then

-- ðŸš€ SUPER SPEED
function ApplySuperSpeed()
    if Config.SuperSpeed then
        localPlayer.Character.Humanoid.WalkSpeed = 120
    else
        localPlayer.Character.Humanoid.WalkSpeed = 16
    end
end

-- ðŸŒŠ ÄI DÆ¯á»šI BIá»‚N
function ApplyUnderwaterWalk()
    if Config.UnderwaterWalk then
        localPlayer.Character.Humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
        localPlayer.Character.Humanoid:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
    end
end

-- TÃŒM QUÃI Gáº¦N NHáº¤T
function FindNearestEnemy()
    local nearest = nil
    local distance = math.huge
    
    if Workspace:FindFirstChild("Enemies") then
        for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
            if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                if enemy.Humanoid.Health > 0 then
                    local dist = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                    if dist < distance then
                        distance = dist
                        nearest = enemy
                    end
                end
            end
        end
    end
    
    return nearest
end

function SafeTeleport(cframe)

    end)
end
