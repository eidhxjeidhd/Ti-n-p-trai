-- --- TAB KAITUN: HỆ THỐNG CÀY THUÊ TỰ ĐỘNG (TỪ A ĐẾN Z) ---
local KaitunTab = Window:CreateTab("Kaitun (A-Z)", 4483362458)
KaitunTab:CreateSection("Hệ Thống Kaitun Max Setting Của Gà Gaming")

KaitunTab:CreateToggle({
   Name = "Bật Kaitun 1 - 2800 (Auto Max Level)",
   CurrentValue = false,
   Callback = function(Value)
      _G.Kaitun_MaxLevel = Value
      if Value then
          Rayfield:Notify({Title = "Gà Gaming Kaitun", Content = "Hệ thống đang tự động định vị Level và quét nhiệm vụ...", Duration = 3})
          -- Logic: Quét Level hiện tại -> Nhận nhiệm vụ -> Bay đến bãi quái -> Đánh
      end
   end,
})

KaitunTab:CreateToggle({
   Name = "Auto Chuyển Biển (Sea 1 -> Sea 4)",
   CurrentValue = false,
   Callback = function(Value)
      _G.Kaitun_AutoSea = Value
      -- Logic: Đủ Level 700 tự đi giải đố mở Sea 2, đủ 1500 tự mở Sea 3...
   end,
})

KaitunTab:CreateToggle({
   Name = "Kaitun All Melee (Tự mua toàn bộ Võ)",
   CurrentValue = false,
   Callback = function(Value)
      _G.Kaitun_Melee = Value
      -- Logic: Tự động gom tiền, đủ điểm Beli/Fragment tự bay đến NPC mua God Human, Sanguine Art...
   end,
})

KaitunTab:CreateSection("Quản Lý Tiền Khí Cụ & Trái Ác Quỷ")

KaitunTab:CreateToggle({
   Name = "Auto Random Trái Ác Quỷ (Cất vào rương)",
   CurrentValue = false,
   Callback = function(Value)
      _G.Kaitun_RandomFruit = Value
      -- Logic: Cứ mỗi 2 tiếng tự động ra NPC Random, được trái gì cũng tự Store vào rương
   end,
})

KaitunTab:CreateToggle({
   Name = "Auto Thức Tỉnh (Awakening) Trái Đang Dùng",
   CurrentValue = false,
   Callback = function(Value)
      _G.Kaitun_Awake = Value
      -- Logic: Tự động mua chip, đi Raid và thức tỉnh chiêu thức
   end,
})

KaitunTab:CreateButton({
   Name = "Kích Hoạt Chế Độ Cày Thuê Xuyên Đêm",
   Callback = function()
       -- Tự động bật màn hình trắng + Bật tất cả các tính năng Kaitun
       game:GetService("RunService"):Set3dRenderingEnabled(false)
       _G.Kaitun_MaxLevel = true
       _G.Kaitun_AutoSea = true
       _G.Kaitun_Melee = true
       Rayfield:Notify({Title = "Chế Độ Cày Thuê", Content = "Đã tối ưu CPU cho máy. Sẵn sàng treo xuyên đêm!", Duration = 5})
   end,
})
