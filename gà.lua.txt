    if state then
        spawn(function()
            while getgenv().RedzSettings.AutoFarm and getgenv().RedzSettings.AutoFarm_Level do
                pcall(function()
                    local Target
                    local Distance = 5000
                    
                    for i,v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
                        if v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                            local Mag = (Player.Character.HumanoidRootPart.Position - v.HumanoidRootPart.Position).Magnitude
                            if Mag < Distance then
                                Distance = Mag
                                Target = v
                            end
                        end
                    end
                    
                    if Target then
                        -- Fast Attack nh∆∞ Redz Hub
                        if getgenv().RedzSettings.FastAttack then
                            for i = 1, 3 do
                                game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Click")
                            end
                        else
                            game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Click")
                        end
                        
                        -- Bring Mob nh∆∞ Redz Hub
                        if getgenv().RedzSettings.BringMob then
                            Target.HumanoidRootPart.CFrame = Player.Character.HumanoidRootPart.CFrame * CFrame.new(0, 0, -20)
                        end
                        
                        -- Di chuy·ªÉn ƒë·∫øn target
                        Player.Character.HumanoidRootPart.CFrame = Target.HumanoidRootPart.CFrame * CFrame.new(0, 20, 10)
                    end
                end)

            end
        end)
    end
end)
