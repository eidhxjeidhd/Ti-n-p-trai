local function toggleFeature(featureName, category)
    local feature = Features
    for _, cat in ipairs(category) do
        feature = feature[cat]
    end
    
    feature.Enabled = not feature.Enabled
    print(string.format("[%s] %s: %s", 
        table.concat(category, "."), 
        featureName, 
        feature.Enabled and "ENABLED" or "DISABLED"
    ))
    
    if feature.Enabled then
        startFeature(featureName, category)
    else
        stopFeature(featureName, category)
    end
end
