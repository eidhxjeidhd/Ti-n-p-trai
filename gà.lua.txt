-- HÃ m Auto Farm
function StartAutoFarm()
    spawn(function()
        while wait(0.5) do
            if not Config.AutoFarm then break end
            
            -- TÃ¬m quÃ¡i gáº§n nháº¥t
            local target = nil
            local shortestDistance = 100
            
            if Workspace:FindFirstChild("Enemies") then
                for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
                    if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                        if enemy.Humanoid.Health > 0 then
                            local distance = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                            if distance < shortestDistance then
                                shortestDistance = distance
                                target = enemy
                            end
                        end
                    end
                end
            end
            
            -- Náº¿u tÃ¬m tháº¥y quÃ¡i
            if target then
                -- Teleport Ä‘áº¿n quÃ¡i
                localPlayer.Character.HumanoidRootPart.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                
                -- Auto Ä‘Ã¡nh
                if Config.AutoMelee then
                    VirtualInput:SendKeyEvent(true, "X", false, game)
                    wait(0.1)
                    VirtualInput:SendKeyEvent(false, "X", false, game)
                end
                
                if Config.AutoSword then
                    VirtualInput:SendKeyEvent(true, "Z", false, game)
                    wait(0.1)
                    VirtualInput:SendKeyEvent(false, "Z", false, game)
                end
            end
        end
    end)
end

-- HÃ m Super Speed
function ApplySuperSpeed()
    if Config.SuperSpeed then
        localPlayer.Character.Humanoid.WalkSpeed = 120
    else
        localPlayer.Character.Humanoid.WalkSpeed = 16
    end
end

-- HÃ m Fly
function ApplyFly()
    if Config.Fly then
        localPlayer.Character.Humanoid:ChangeState(Enum.HumanoidStateType.Flying)
    end
end

-- Cháº¡y cÃ¡c tÃ­nh nÄƒng
spawn(function()
    while wait(1) do
        if Config.AutoFarm then StartAutoFarm() end
        ApplySuperSpeed()
        if Config.Fly then ApplyFly() end
    end
end)
