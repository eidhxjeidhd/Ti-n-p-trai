-- ðŸš€ CÃC HÃ€M CHá»¨C NÄ‚NG THá»°C Táº¾ (Cáº¦N ÄÆ¯á»¢C TRIá»‚N KHAI)

function UltimateAutoFarm()
    spawn(function()
        while task.wait(0.3) do
            if not Config.AutoFarmLevel then break end
            
            local target = FindNearestEnemy()
            if target then
                -- Teleport vÃ  táº¥n cÃ´ng vá»›i tá»‘c Ä‘á»™ cao
                localPlayer.Character.HumanoidRootPart.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                
                -- Auto click cá»±c nhanh
                if Config.AutoClickMelee then
                    for i = 1, 5 do
                        VirtualInput:SendKeyEvent(true, "X", false, game)
                        task.wait(0.05)
                        VirtualInput:SendKeyEvent(false, "X", false, game)
                    end
                end
            end
        end
    end)
end

function ApplyGodMode()
    if Config.GodMode then
        localPlayer.Character.Humanoid.Health = math.huge
    end
end

function ApplySuperSpeed()
    if Config.SuperSpeed300 then
        localPlayer.Character.Humanoid.WalkSpeed = 120
    else
        localPlayer.Character.Humanoid.WalkSpeed = 16
    end
end

function FindNearestEnemy()
    local nearest = nil
    local shortestDistance = math.huge
    
    if Workspace:FindFirstChild("Enemies") then
        for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
            if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                if enemy.Humanoid.Health > 0 then
                    local dist = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                    if dist < shortestDistance then
                        shortestDistance = dist
                        nearest = enemy
                    end
                end
            end
        end
    end
    
    return nearest
end
