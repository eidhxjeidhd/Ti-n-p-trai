-- AUTO FARM LEVEL
function StartAutoFarm()
    spawn(function()
        while wait(0.5) do
            if not Config.AutoFarm then 
                print("ðŸ›‘ Auto Farm Ä‘Ã£ dá»«ng")
                break 
            end
            
            local target = FindNearestEnemy()
            if target then
                -- TELEPORT Äáº¾N QUÃI
                localPlayer.Character.HumanoidRootPart.CFrame = target.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                
                -- AUTO CLICK Táº¤N CÃ”NG
                if Config.AutoClick then
                    VirtualInput:SendKeyEvent(true, "X", false, game)
                    wait(0.1)
                    VirtualInput:SendKeyEvent(false, "X", false, game)
                    
                    VirtualInput:SendKeyEvent(true, "Z", false, game)
                    wait(0.1)
                    VirtualInput:SendKeyEvent(false, "Z", false, game)
                end
                
                -- AIMBOT
                if Config.Aimbot then
                    localPlayer.Character.HumanoidRootPart.CFrame = CFrame.lookAt(
                        localPlayer.Character.HumanoidRootPart.Position,
                        target.HumanoidRootPart.Position
                    )
                end
                
                print("ðŸŽ¯ Äang farm: " .. target.Name)
            end
        end
    end)
end

-- AUTO BOSS
function StartAutoBoss()
    spawn(function()
        while wait(2) do
            if not Config.AutoBoss then break end
            
            for _, boss in pairs(Workspace.Enemies:GetChildren()) do
                if string.find(boss.Name, "Boss") or string.find(boss.Name, "Saber") then
                    if boss:FindFirstChild("HumanoidRootPart") and boss.Humanoid.Health > 0 then
                        localPlayer.Character.HumanoidRootPart.CFrame = boss.HumanoidRootPart.CFrame * CFrame.new(0, 0, 8)
                        print("ðŸ‘‘ Äang Ä‘Ã¡nh boss: " .. boss.Name)
                        break
                    end
                end
            end
        end
    end)
end

    end
end

-- TÃŒM QUÃI Gáº¦N NHáº¤T
function FindNearestEnemy()
    local nearest = nil
    local distance = math.huge
    
    if Workspace:FindFirstChild("Enemies") then
        for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
            if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                if enemy.Humanoid.Health > 0 then
                    local dist = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                    if dist < distance then
                        distance = dist
                        nearest = enemy
                    end
                end
            end
        end
    end
    
    return nearest
end
