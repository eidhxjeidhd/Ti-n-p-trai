-- Ki·ªÉm tra an to√†n
function safetyCheck()
    if not localPlayer.Character or not localPlayer.Character:FindFirstChild("Humanoid") or localPlayer.Character.Humanoid.Health <= 0 then
        return false
    end
    return true
end

-- T√¨m enemy g·∫ßn nh·∫•t
function findNearestEnemy()
    local nearestEnemy = nil
    local shortestDistance = math.huge
    
    for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
        if enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
            local distance = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
            if distance < shortestDistance then
                shortestDistance = distance
                nearestEnemy = enemy
            end
        end
    end
    
    return nearestEnemy
end

-- Thu th·∫≠p item r∆°i
function collectDrops()
    for _, drop in pairs(Workspace:GetChildren()) do
        if drop.Name == "Drop" and drop:FindFirstChild("Handle") then
            localPlayer.Character.HumanoidRootPart.CFrame = drop.Handle.CFrame
            wait(0.3)
        end
    end
end

-- Auto Farm v·ªõi Teleport
function autoFarmWithTeleport()
    while wait(0.5) do
        if not _G.AutoFarm then break end
        if not safetyCheck() then continue end
        
        local enemy = findNearestEnemy()
        if enemy then
            if _G.AutoTeleport then
                localPlayer.Character.HumanoidRootPart.CFrame = enemy.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
            end
            
            game:GetService("VirtualInputManager"):SendKeyEvent(true, "X", false, game)
            wait(0.1)
            game:GetService("VirtualInputManager"):SendKeyEvent(false, "X", false, game)
            
            if _G.AutoCollectDrops then
                collectDrops()
            end
        end
    end
end

-- Auto Buy Devil Fruits
function buyDevilFruits()
    local fruitShop = Workspace:FindFirstChild("FruitShop")
    if fruitShop then
        for _, fruit in pairs(fruitShop:GetChildren()) do
            if fruit:IsA("Model") and fruit:FindFirstChild("Handle") then
                Remotes.CommF_:InvokeServer("PurchaseRawFruit", fruit.Name)
                wait(0.5)
            end
        end
    end
end

-- Auto Buy Swords
function buySwords()
    local swordShop = Workspace:FindFirstChild("SwordShop")
    if swordShop then
        for _, sword in pairs(swordShop:GetChildren()) do
            if sword:IsA("Model") and sword:FindFirstChild("Handle") then
                Remotes.CommF_:InvokeServer("BuyItem", sword.Name)
                wait(0.5)
            end
        end
    end
end

-- Auto Farm Bones
function farmBones()
    while wait(1) do
        if not _G.AutoFarmBones then break end
        if not safetyCheck() then continue end
        
        for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
            if enemy.Name:lower():find("skeleton") and enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
                localPlayer.Character.HumanoidRootPart.CFrame = enemy.HumanoidRootPart.CFrame
                wait(0.3)
                game:GetService("VirtualInputManager"):SendKeyEvent(true, "X", false, game)
            end
        end
    end
end
