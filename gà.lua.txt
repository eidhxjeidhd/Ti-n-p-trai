-- T·∫°o c√°c tab v·ªõi style ƒë·∫πp h∆°n
local function createTabButton(tabName, position)
    local button = Instance.new("TextButton")
    button.Name = tabName .. "Tab"
    button.Parent = TabButtons
    button.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
    button.BorderSizePixel = 0
    button.Position = position
    button.Size = UDim2.new(0.9, 0, 0, 45)
    button.Font = Enum.Font.GothamBold
    button.Text = Tabs[tabName].Name
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.TextSize = 14
    
    local buttonCorner = Instance.new("UICorner")
    buttonCorner.CornerRadius = UDim.new(0.1, 0)
    buttonCorner.Parent = button
    
    local buttonStroke = Instance.new("UIStroke")
    buttonStroke.Parent = button
    buttonStroke.Color = Color3.fromRGB(100, 100, 150)
    buttonStroke.Thickness = 1
    
    button.MouseButton1Click:Connect(function()
        currentTab = tabName
        updateContent()
    end)
    
    -- Hi·ªáu ·ª©ng hover
    button.MouseEnter:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
    end)
    
    button.MouseLeave:Connect(function()
        if currentTab == tabName then
            button.BackgroundColor3 = Color3.fromRGB(70, 130, 200)
        else
            button.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
        end
    end)
    
    return button
end

-- C·∫≠p nh·∫≠t m√†u tab active
local function updateContent()
    for tabName, tabData in pairs(Tabs) do
        if tabData.Button then
            if tabName == currentTab then
                tabData.Button.BackgroundColor3 = Color3.fromRGB(70, 130, 200)
            else
                tabData.Button.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
            end
        end
        if tabData.Content then
            tabData.Content.Visible = (tabName == currentTab)
        end
    end
end
