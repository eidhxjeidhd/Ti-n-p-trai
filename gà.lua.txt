-- Auto Farm Function
function StartAutoFarm()
    spawn(function()
        while wait() do
            if not getgenv().GaGaming.AutoFarm then break end
            
            -- TÃ¬m enemy gáº§n nháº¥t
            local target = FindNearestEnemy()
            if target then
                -- Teleport Ä‘áº¿n enemy
                if getgenv().GaGaming.AutoTeleport then
                    SafeTeleport(target.HumanoidRootPart.CFrame * CFrame.new(0, 0, 8))
                end
                
                -- Táº¥n cÃ´ng
                AttackEnemy(target)
            end
        end
    end)
end

-- TÃ¬m enemy gáº§n nháº¥t
function FindNearestEnemy()
    local nearest = nil
    local distance = math.huge
    
    for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
        if enemy:FindFirstChild("Humanoid") and enemy.Humanoid.Health > 0 then
            local dist = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
            if dist < distance then
                distance = dist
                nearest = enemy
            end
        end
    end
    
    return nearest
end

-- Teleport an toÃ n
function SafeTeleport(cframe)
    localPlayer.Character.HumanoidRootPart.CFrame = cframe
end

function AttackEnemy(enemy)

    wait(0.1)

end

function StartCollectDrops()
    spawn(function()
        while wait(3) do
            if not getgenv().GaGaming.AutoCollectDrops then break end
            
            for _, drop in pairs(Workspace:GetChildren()) do
                if drop.Name == "Drop" and drop:FindFirstChild("Handle") then
                    SafeTeleport(drop.Handle.CFrame)
                    wait(0.5)
                end
            end
        end
    end)
end
