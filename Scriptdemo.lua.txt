-- AUTO FARM LEVEL
function RealAutoFarmLevel()
    spawn(function()
        while wait(0.5) do
            if not Config.AutoFarmLevel then break end
            
            local target = FindNearestEnemy()
            if target then
                SafeTeleport(target.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5))
                
                -- Auto attack
                if Config.AutoMelee then
                    VirtualInput:SendKeyEvent(true, "X", false, game)
                    wait(0.1)
                    VirtualInput:SendKeyEvent(false, "X", false, game)
                end
                
                if Config.AutoSword then
                    VirtualInput:SendKeyEvent(true, "Z", false, game)
                    wait(0.1)
                    VirtualInput:SendKeyEvent(false, "Z", false, game)
                end
            end
        end
    end)
end

-- AUTO FARM BOSSES
function RealAutoFarmBosses()
    spawn(function()
        while wait(2) do
            if not Config.AutoFarmBosses then break end
            
            for _, boss in pairs(Workspace.Enemies:GetChildren()) do
                if string.find(boss.Name, "Boss") then
                    if boss:FindFirstChild("HumanoidRootPart") and boss.Humanoid.Health > 0 then
                        SafeTeleport(boss.HumanoidRootPart.CFrame * CFrame.new(0, 0, 10))
                        break
                    end
                end
            end
        end
    end)
end

-- AUTO COLLECT DROPS
function RealAutoCollectDrops()
    spawn(function()
        while wait(3) do
            if not Config.AutoCollectDrops then break end
            
            for _, item in pairs(Workspace:GetChildren()) do
                if item.Name == "Drop" and item:FindFirstChild("Handle") then
                    SafeTeleport(item.Handle.CFrame)
                    wait(0.5)
                end
            end
        end
    end)
end

-- SUPER SPEED
function RealSuperSpeed()

        localPlayer.Character.Humanoid.WalkSpeed = 120

        localPlayer.Character.Humanoid.WalkSpeed = 16
    end
end

-- FLY
function RealFly()
    if Config.Fly then
        localPlayer.Character.Humanoid:ChangeState(Enum.HumanoidStateType.Flying)
    end
end

-- TÃŒM QUÃI Gáº¦N NHáº¤T
function FindNearestEnemy()
    local nearest = nil
    local distance = math.huge
    
    if Workspace:FindFirstChild("Enemies") then
        for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
            if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                if enemy.Humanoid.Health > 0 then
                    local dist = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                    if dist < distance then
                        distance = dist
                        nearest = enemy
                    end
                end
            end
        end
    end
    
    return nearest
end

-- TELEPORT AN TOÃ€N
function SafeTeleport(cframe)
    pcall(function()
        localPlayer.Character.HumanoidRootPart.CFrame = cframe
    end)
end
