-- üöÄ AUTO FARM LEVEL - HO·∫†T ƒê·ªòNG TH·∫¨T
function StartAutoFarm()
    spawn(function()
        while wait(1) do
            if not Config.AutoFarm then 
                StatusLabel.Text = "üî¥ Auto Farm ƒë√£ t·∫Øt"
                break 
            end
            
            StatusLabel.Text = "üü¢ ƒêang Auto Farm Level..."
            
            -- T√åM QU√ÅI G·∫¶N NH·∫§T
            local nearestEnemy = nil
            local shortestDistance = 100
            
            if Workspace:FindFirstChild("Enemies") then
                for _, enemy in pairs(Workspace.Enemies:GetChildren()) do
                    if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
                        if enemy.Humanoid.Health > 0 then
                            local distance = (localPlayer.Character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
                            if distance < shortestDistance then
                                shortestDistance = distance
                                nearestEnemy = enemy
                            end
                        end
                    end
                end
            end
            
            -- N·∫æU T√åM TH·∫§Y QU√ÅI
            if nearestEnemy then
                StatusLabel.Text = "üéØ ƒêang farm: " .. nearestEnemy.Name
                
                -- TELEPORT N·∫æU B·∫¨T
                if Config.AutoTeleport then
                    localPlayer.Character.HumanoidRootPart.CFrame = nearestEnemy.HumanoidRootPart.CFrame * CFrame.new(0, 0, 5)
                    wait(0.5)
                end
                
                -- T·∫§N C√îNG
                VirtualInput:SendKeyEvent(true, "X", false, game)
                wait(0.1)
                VirtualInput:SendKeyEvent(false, "X", false, game)
                
            else
                StatusLabel.Text = "üîç ƒêang t√¨m qu√°i..."
            end
        end
    end)
end
