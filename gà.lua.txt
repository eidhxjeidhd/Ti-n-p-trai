            while getgenv().AutoFarmLevel do
                wait(.1)
                pcall(function()
                    local MaxDistance = math.huge
                    local Target
                    for i,v in pairs(game:GetService("Workspace").Enemies:GetChildren()) do
                        if v:FindFirstChild("Humanoid") and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                            local Distance = (Player.Character.HumanoidRootPart.Position - v.HumanoidRootPart.Position).Magnitude
                            if Distance < MaxDistance then
                                Target = v
                                MaxDistance = Distance
                            end
                        end
                    end
                    if Target then
                        Player.Character.HumanoidRootPart.CFrame = Target.HumanoidRootPart.CFrame * CFrame.new(0, 20, 5)
                        game:GetService("ReplicatedStorage").Remotes.CommF_:InvokeServer("Click")
                    end
                end)
            end
        end)
    end
end)
